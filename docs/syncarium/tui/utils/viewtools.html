  95    def console_message(
  96        self,
  97        type: str,
  98        message: str,
  99        title_emoji: str = "🔷",
 100        indent: int = 0,
 101        logger: Optional[logging.Logger] = None
 102    ) -> None:
 103        """
 104        Displays a styled message in the console with optional indentation and logging.
 105
 106        Applies visual formatting based on message type, including emoji and color style.
 107        Supports logging the plain-text version of the message if a logger is provided.
 108
 109        ### Args
 110        - **type** (`str`): Type of message (e.g., `"success"`, `"error"`, `"info"`).
 111        - **message** (`str`): Content of the message to display.
 112        - **title_emoji** (`str`): Emoji used for title-type messages. Defaults to `"🔷"`.
 113        - **indent** (`int`): Indentation level for visual hierarchy. Defaults to `0`.
 114        - **logger** (`Optional[Logger]`): Logger instance to log the plain-text message. Defaults to `None`.
 115        """
 116
 117        # Define styles and emojis for different message types
 118        styles = {
 119            "success": ("✅", "[green]"),
 120            "error": ("❌", "[red]"),
 121            "caution": ("⚠️", "[yellow]"),
 122            "info": ("ℹ️", "[italic cyan]"),
 123            "clean": ("🗑️", "[italic bright_black]"),
 124            "exit": ("👋", "[bright_blue]"),
 125            "main_title": (f"{title_emoji}", "[bold underline bright_white]"),
 126            "title": (f"{title_emoji}", "[bold blue]")
 127        }
 128
 129        # Get emoji and style based on message type
 130        emoji, style = styles.get(type.lower(), ("❔", "[white]"))
 131
 132        # Create visual indentation prefix
 133        prefix = "" if indent == 0 else "│   " * (indent - 1) + "├── "
 134
 135        # Format each line with style and indentation
 136        lines = message.split("\n")
 137        formatted_lines = [
 138            f"{prefix}{emoji} {style}{line}{style.replace('[', '[/')}"
 139            for line in lines
 140        ]
 141        formatted_message = "\n".join(formatted_lines)
 142
 143        if logger:
 144            # Log plain text version if logger is provided
 145            plain_lines = [f"{prefix}{emoji} {line}" for line in lines]
 146            plain_message = "\n".join(plain_lines)
 147            log_func = getattr(logger, type.lower(), logger.info)
 148            log_func(plain_message)
 149        else:
 150            # Print styled message to console
 151            self.console.print(formatted_message)

 163        Displays a stylized ASCII art title centered in the terminal, with optional version string.

 165        Clears the console, renders the title using `pyfiglet`, and centers it horizontally.

 167

 176        # Create ASCII art
 177        ascii_art = pyfiglet.figlet_format(text, font=font)
 178        # Get terminal width
 179        terminal_width = shutil.get_terminal_size().columns
 180        for line in ascii_art.splitlines():
 181            padding = max((terminal_width - len(line)) // 2, 0)
 182            self.console.print(" " * padding + line, style="bold cyan")
 183            time.sleep(delay)
 184
 185        # Mostrar la versión justo debajo, centrada y en estilo más pequeño
 186        if self.version:
 187            version_text = f"v{self.version}"
 188            padding = max((terminal_width - len(version_text)) // 2, 0)

 228        """
 229        Displays a styled selection menu in the console with visual indentation for choices.

 232        display of options and accepts both strings and dictionaries with `name` and `value`.
 233
 234        ### Args
 235        - **choices** (`list`): List of options to display. Can be strings or dicts with `name` and `value`.
 236        - **message** (`str`): Prompt message shown above the menu. Defaults to `"Select an option:"`.
 237        - **indent** (`int`): Indentation level for visual hierarchy. Defaults to `0`.
 238
 239        ### Returns
 240        - `Any`: The selected value from the menu.
 241        """
 242
 243        inquirer_custom_style = get_style({
 244            "question": "fg:#00ffff italic",
 245            "questionmark": "fg:#ff00ff ",
 246            "answer": "fg:#4169e1 ",
 247            "pointer": "fg:#00ffff ",
 248            "highlighted": "#008b8b ",
 249            "selected": "#008b8b ",
 250            "instruction": "fg:#cccccc ",
 251            "text": "fg:#ffffff ",
 252            "disabled": "fg:#666666 italic ",
 253        })
 254
 255        # Visual indentation prefix for choices
 256        def get_indent_prefix(level):
 257            if level == 0:
 258                return ""
 259            return "│   " * (level - 1) + "├── "
 260
 261        prefix = get_indent_prefix(indent)
 262
 263        # Leave the message clean
 264        indented_message = message
 265
 266        # Apply visual indentation only to choices

 273                })
 274            else:
 275                indented_choices.append(f"{prefix}{choice}")

 277        question = inquirer.select(
 278            message=indented_message,
 279            choices=indented_choices,

 282            pointer="👉",
 283            cycle=True,

 343# ─────────────────────────────────────────────────────────────────────────────
 344    def real_time_progress(
 345        self,
 346        start_time: float,
 347        duration: int,
 348        message: str,
 349        repetition_info: Optional[int]
 350    ) -> None:
 351        """
 352        Displays a real-time progress bar for the duration of a running process.
 353
 354        Uses Rich's progress bar to visually represent elapsed time. Updates until the
 355        specified duration is reached or the user interrupts the display.
 356
 357        ### Args
 358        - **start_time** (`float`): Timestamp when the process started (in seconds since epoch).
 359        - **duration** (`int`): Total duration of the process in seconds.
 360        - **message** (`str`): Message to display alongside the progress bar.
 361        - **repetition_info** (`Optional[int]`): Optional repetition count to append to the message.
 362        """
 363
 364        if not start_time or not duration:
 365            self.console_message("error", "Process has not started yet.")
 366            return
 367
 368        total_seconds = duration
 369        elapsed = lambda: time.time() - start_time
 370
 371        try:
 372            with Progress(
 373                "[progress.description]{task.description}",
 374                BarColumn(),
 375                "[progress.percentage]{task.percentage:>3.0f}%",
 376                TimeRemainingColumn(),
 377                console=self.console,
 378                transient=True
 379            ) as progress:
 380                message = message + (f" ({repetition_info})" if repetition_info else "")
 381                task = progress.add_task(message, total=total_seconds)
 382                self.console_message("info", "Press Ctrl+C to exit.")
 383
 384                # Update progress bar until time is up or user interrupts
 385                while not progress.finished:
 386                    progress.update(task, completed=elapsed())
 387                    if elapsed() >= total_seconds:
 388                        break
 389                    time.sleep(0.5)
 390
 391            self.console_message("success", "Progress completed.")
 392        except KeyboardInterrupt:
 393            self.console_message("caution", "Progress display cancelled by user.")

 399    def table_network_namespaces(self) -> None:
 400        """
 401        Displays a table of network namespaces and their associated interfaces.
 402
 403        Uses system commands to retrieve information about existing network namespaces,
 404        including their interfaces, IP addresses, MAC addresses, and default gateways.
 405        The data is presented in a styled table using the `rich` library. If no namespaces
 406        are found, a warning panel is displayed.
 407        """
 408
 409        try:
 410            # Get the list of network namespaces
 411            result = subprocess.run(["sudo","ip", "netns", "list"], capture_output=True, text=True, check=True)
 412            namespaces = result.stdout.strip().split('\n')
 413
 414            # Create the table layout
 415            table = Table(
 416                title="🌐 Network Namespaces",
 417                show_header=True,
 418                header_style="bold cyan",
 419                title_justify="left"
 420            )
 421            table.add_column("Namespace", justify="center", overflow="fold")
 422            table.add_column("Interface", justify="center", overflow="fold")
 423            table.add_column("IP Address", justify="center", overflow="fold")
 424            table.add_column("MAC Address", justify="center", overflow="fold")
 425            table.add_column("Gateway", justify="center", overflow="fold")
 426
 427            if namespaces and namespaces[0]:
 428                for ns_line in namespaces:
 429                    ns = ns_line.split()[0]
 430                    try:
 431                        # Get interfaces in the namespace
 432                        interfaces_result = subprocess.run(
 433                            ["sudo", "ip", "netns", "exec", ns, "ip", "-o", "link", "show"],
 434                            capture_output=True, text=True, check=True
 435                        )
 436                        interfaces = interfaces_result.stdout.strip().split('\n')
 437                        if not interfaces:
 438                            continue
 439
 440                        # Get IP addresses
 441                        try:
 442                            ip_result = subprocess.run(
 443                                ["sudo", "ip", "netns", "exec", ns, "ip", "-o", "-4", "addr", "show"],
 444                                capture_output=True, text=True, check=True
 445                            )
 446                            ips = ip_result.stdout.strip().split('\n')
 447                        except subprocess.CalledProcessError:
 448                            ips = []
 449
 450                        # Get default gateways
 451                        try:
 452                            gw_result = subprocess.run(
 453                                ["sudo", "ip", "netns", "exec", ns, "ip", "route", "show", "default"],
 454                                capture_output=True, text=True, check=True
 455                            )
 456                            gws = gw_result.stdout.strip().split('\n')
 457                        except subprocess.CalledProcessError:
 458                            gws = []
 459
 460                        for line in interfaces:
 461                            parts = line.split()
 462                            if len(parts) >= 2:
 463                                iface = parts[1].strip(':')
 464                                if iface == "lo":
 465                                    continue
 466                                # Get MAC address
 467                                try:
 468                                    mac_result = subprocess.run(
 469                                        ["sudo", "ip", "netns", "exec", ns, "ip", "link", "show", iface],
 470                                        capture_output=True, text=True, check=True
 471                                    )
 472                                    mac_line = next((l for l in mac_result.stdout.splitlines() if "link/ether" in l), None)
 473                                    mac = mac_line.split()[1] if mac_line else "N/A"
 474                                except subprocess.CalledProcessError:
 475                                    mac = "N/A"
 476
 477                                # Match IP and gateway to interface
 478                                ip = next((ip_line.split()[3] for ip_line in ips if iface in ip_line), "N/A")
 479                                gw = next((gw_line.split()[2] for gw_line in gws if "via" in gw_line), "N/A")
 480
 481                                # Add row to the table
 482                                table.add_row(ns, iface, ip, mac, gw)
 483
 484                    except subprocess.CalledProcessError as e:
 485                        # Handle errors per namespace
 486                        self.console_message("error",f"Error processing namespace '{ns}': {e}")
 487                        continue
 488
 489                # Display the final table
 490                self.console.print(table)
 491
 492            else:
 493                # No namespaces found
 494                self.console.print(Panel.fit(
 495                    "[bold yellow]No network namespaces found.[/bold yellow]",
 496                    title="🌐 Network Namespaces",
 497                    border_style="red"
 498                ))
 499
 500        except subprocess.CalledProcessError as e:
 501            # Handle general error
 502            self.console_message("error",f"Error retrieving namespaces: {e}")

 507    def table_synccore_processes(self) -> None:
 508        """
 509        Displays a table of currently running PTP (Precision Time Protocol) processes.
 510
 511        Uses `pgrep` and `ps` to identify running `ptp4l` processes and extract details
 512        such as PID, start time, command line, and stdout redirection path. If no processes
 513        are found, a warning panel is shown.
 514        """
 515
 516        # Run pgrep to find all running ptp4l processes with full command line
 517        result = subprocess.run(["sudo","pgrep", "-a", "ptp4l"], capture_output=True, text=True)
 518        lines = result.stdout.strip().splitlines()
 519
 520        # If no processes are found, show a warning panel
 521        if not lines:
 522            self.console.print(Panel.fit(
 523                "[bold yellow]No PTP processes have been started or are currently running.[/bold yellow]",
 524                title="⏱️ PTP Processes Status",
 525                border_style="red"
 526            ))
 527            return
 528
 529        # Create a rich table to display process information
 530        table = Table(
 531            title="⏱️ PTP Processes Status",
 532            show_header=True,
 533            header_style="bold cyan",
 534            title_justify="left",
 535        )
 536        table.add_column("PID", justify="center", overflow="fold")
 537        table.add_column("Start Time", justify="center", overflow="fold")
 538        table.add_column("Command", justify="left", overflow="fold")
 539        table.add_column("Stdout Redirection", justify="left", overflow="fold")
 540
 541        # Populate the table with process details
 542        for line in lines:
 543            pid, *cmd = line.split()
 544            cmd_str = " ".join(cmd)
 545
 546            # Get process start time
 547            start_time = datetime.datetime.strptime(
 548                subprocess.run(
 549                    ["sudo", "ps", "-p", pid, "-o", "lstart="], capture_output=True, text=True
 550                ).stdout.strip().replace("  ", " "), "%a %b %d %H:%M:%S %Y"
 551            ).strftime("%d-%m-%Y %H:%M:%S")
 552
 553
 554
 555            # Get stdout redirection path (if any)
 556            stdout_path = subprocess.run(
 557                ["sudo", "readlink", f"/proc/{pid}/fd/1"],
 558                capture_output=True,
 559                text=True
 560            ).stdout.strip() or "N/A"
 561
 562            table.add_row(pid, start_time, cmd_str, stdout_path)
 563
 564        # Print the table to the console
 565        self.console.print(table)

 570    def table_dpdk_bound_devices(self, dpdk_path: str) -> List[str]:
 571        """
 572        Lists network devices currently bound to DPDK-compatible drivers.
 573
 574        Executes the `dpdk_nic_bind.py -s` script to retrieve information about
 575        network interfaces using DPDK drivers. Parses the output to extract device
 576        details and displays them in a formatted Rich table.
 577
 578        ### Args
 579        - **dpdk_path** (`str`): Path to the directory containing the `dpdk_devbind.py` script.
 580
 581        ### Returns
 582        - `List[str]`: List of strings representing DPDK-bound devices, or an empty list if none are found.
 583        """
 584
 585        try:
 586            # Run the dpdk_nic_bind.py script with the -s flag to list devices
 587            result = subprocess.run(
 588                ["sudo", "./dpdk-devbind.py", "-s"],
 589                capture_output=True,
 590                text=True,
 591                check=True,
 592                cwd=dpdk_path  # Set working directory to where the script is located
 593            )
 594            output = result.stdout
 595
 596            dpdk_section = []
 597            capture = False
 598
 599            # Parse the output to isolate the section listing DPDK-bound devices
 600            for line in output.splitlines():
 601                if "Network devices using DPDK-compatible driver" in line:
 602                    capture = True
 603                    continue
 604                if capture:
 605                    if line.strip() == "" or line.startswith("==="):
 606                        continue
 607                    if line.startswith("Network devices using kernel driver"):
 608                        break
 609                    dpdk_section.append(line.strip())
 610
 611            dpdk_devices = []
 612            pattern = re.compile(
 613                r"(?P<pci>\S+)\s+'(?P<name>[^']+)'.*?drv=(?P<driver>\S+)\s+unused=(?P<unused>\S+)"
 614            )
 615
 616            for line in dpdk_section:
 617                match = pattern.search(line)
 618                if match:
 619                    dpdk_devices.append(match)
 620

 345    def real_time_progress(
 346        self,
 347        start_time: float,
 348        duration: int,
 349        message: str,
 350        repetition_info: Optional[int]
 351    ) -> None:
 352        """
 353        Displays a real-time progress bar for the duration of a running process.
 354
 355        Uses Rich's progress bar to visually represent elapsed time. Updates until the
 356        specified duration is reached or the user interrupts the display.
 357
 358        ### Args
 359        - **start_time** (`float`): Timestamp when the process started (in seconds since epoch).
 360        - **duration** (`int`): Total duration of the process in seconds.
 361        - **message** (`str`): Message to display alongside the progress bar.
 362        - **repetition_info** (`Optional[int]`): Optional repetition count to append to the message.
 363        """
 364
 365        if not start_time or not duration:
 366            self.console_message("error", "Process has not started yet.")
 367            return
 368
 369        total_seconds = duration
 370        elapsed = lambda: time.time() - start_time
 371
 372        try:
 373            with Progress(
 374                "[progress.description]{task.description}",
 375                BarColumn(),
 376                "[progress.percentage]{task.percentage:>3.0f}%",
 377                TimeRemainingColumn(),
 378                console=self.console,
 379                transient=True
 380            ) as progress:
 381                message = message + (f" ({repetition_info})" if repetition_info else "")
 382                task = progress.add_task(message, total=total_seconds)
 383                self.console_message("info", "Press Ctrl+C to exit.")
 384
 385                # Update progress bar until time is up or user interrupts
 386                while not progress.finished:
 387                    progress.update(task, completed=elapsed())
 388                    if elapsed() >= total_seconds:
 389                        break
 390                    time.sleep(0.5)
 391
 392            self.console_message("success", "Progress completed.")
 393        except KeyboardInterrupt:
 394            self.console_message("caution", "Progress display cancelled by user.")
