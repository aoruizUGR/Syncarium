152    def _comm(self, code: str, fields: List[str] = [], tries: int = 2) -> List[str]:
153        """
154        Sends a command to the PPS device over UART and parses the response.
155
156        Constructs a command string with checksum, sends it via UART, and validates
157        the response. Retries the communication if an error occurs.
158
159        ### Args
160        - **code** (`str`): Command code to send.
161        - **fields** (`List[str]`, optional): List of fields to include in the command.
162        - **tries** (`int`, optional): Number of retry attempts in case of failure. Defaults to 2.
163
164        ### Returns
165        - **List[str]**: Parsed fields from the device response.
166
167        ### Raises
168        - **Exception**: If communication fails or response is invalid after all retries.
169        """
170
171        for attempt in range(tries):
172            try:
173                # Construct the command string with checksum
174                payload = f"{code}{',' if fields else ''}{','.join(fields)}"
175                checksum = self._checksum(payload)
176                tx = f"${payload}*{checksum}\r\n"
177                self.uart.write(tx.encode())
178
179                # Read and decode the response
180                rx = self.uart.readline().decode().strip()
181                if not rx.startswith('$'):
182                    raise Exception("Invalid response format")
183
184                # Split and validate checksum
185                content, received_chk = rx[1:].split("*")
186                if received_chk != self._checksum(content):
187                    raise Exception("Checksum mismatch")
188
189                parts = content.split(',')
190                if parts[0] == "ER":
191                    raise Exception("Device returned an error")
192
193                return parts[1:]
194
195            except Exception:
196                if attempt == tries - 1:
197                    raise  # Raise the last exception if all attempts fail
198
199                # Reinitialize UART before retrying
200                self.uart.close()
201                self.uart = serial.Serial(port=self.port, baudrate=1000000, timeout=3)
